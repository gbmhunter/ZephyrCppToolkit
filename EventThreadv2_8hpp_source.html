87 // Get the next timer to expire

90 // See if there is an already expired timer, is so, there is no need to block
91 // on the event queue.
92 if (nextTimerInfo.m_timer != nullptr && nextTimerInfo.m_durationToWaitUs == 0) {
93 nextTimerInfo = m_timerManager.getNextExpiringTimer();
94 LOG_DBG("Timer expired. Timer: %p.", nextTimerInfo.m_timer);
95 // Get the timer event
96 nextTimerInfo.m_timer->updateAfterExpiry();
97 return nextTimerInfo.m_timer->getEvent();
98 }
99
100 // If we get here, we have handled all expired timers.
101 k_timeout_t timeout;
102 if (nextTimerInfo.m_timer != nullptr) {
103 timeout = Z_TIMEOUT_US(nextTimerInfo.m_durationToWaitUs);
104 } else {
105 timeout = K_FOREVER;
106 }
107
108 // Block on message queue until next timer expiry
109 // Create storage in the case we receive an external event
110 EventType rxEvent;
111 int queueRc = k_msgq_get(&m_threadMsgQueue, &rxEvent, timeout);
112 if (queueRc == 0) {
113 // We got a message from the queue, so we can handle it
114 return rxEvent;
115 } else if (queueRc == -EAGAIN) {
116 // Queue timed out, which means we need to handle the timer expiry
117 LOG_DBG("Queue timed out, which means we need to handle the timer expiry.");
118 nextTimerInfo.m_timer->updateAfterExpiry();
119 return nextTimerInfo.m_timer->getEvent();
120 } else if (queueRc == -ENOMSG) {
121 // This means the queue must have been purged
122 __ASSERT(false, "Got -ENOMSG from queue, was not expecting this.");
123 } else {
124 __ASSERT(false, "Got unexpected return code from queue: %d.", queueRc);
125 }
126 __ASSERT(false, "Should not get here.");
