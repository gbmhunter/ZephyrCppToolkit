225 void runEventLoop() {
226 LOG_MODULE_DECLARE(EventThread, ZCT_EVENT_THREAD_LOG_LEVEL);
227 LOG_DBG("%s() called.", __FUNCTION__);
228
229 while (!m_exitEventLoop) {
230 // Get the next timer to expire
231 auto nextTimerInfo = m_timerManager.getNextExpiringTimer();
232
233 // Check for expired timers and call their callbacks
234 while (nextTimerInfo.m_timer != nullptr && nextTimerInfo.m_durationToWaitUs == 0) {
235 LOG_DBG("Timer expired. Timer: %p.", nextTimerInfo.m_timer);
236 // Update the timer after expiry
237 nextTimerInfo.m_timer->updateAfterExpiry();
238
239 // Call the callback
240 const auto& callback = nextTimerInfo.m_timer->getExpiryCallback();
241 if (callback) {
242 callback();
243 // If the callback calls exitEventLoop(), we will exit the event loop
244 // and return from runEventLoop().
245 if (m_exitEventLoop) {
246 return;
247 }
248 }
249
250 // Check for the next expired timer
251 nextTimerInfo = m_timerManager.getNextExpiringTimer();
252 }
253
254 // If we get here, we have handled all expired timers.
255 k_timeout_t timeout;
256 if (nextTimerInfo.m_timer != nullptr) {
257 timeout = Z_TIMEOUT_US(nextTimerInfo.m_durationToWaitUs);
258 } else {
259 timeout = K_FOREVER;
260 }
261
262 // Block on message queue until next timer expiry
263 // Create storage in the case we receive an external event
264 MsgQueueItem msgQueueItem;
265 int queueRc = k_msgq_get(&m_threadMsgQueue, &msgQueueItem, timeout);
266 if (queueRc == 0) {
267 // We got a message from the queue, it will either be an event or a function to run in the context of the event thread.
268 if (std::holds_alternative<EventType>(msgQueueItem)) {
269 // It's an event, call the external event callback
270 const auto& event = std::get<EventType>(msgQueueItem);
271 if (m_externalEventCallback) {
272 m_externalEventCallback(event);
273 // If the callback calls exitEventLoop(), we will exit the event loop
274 // and return from runEventLoop().
275 if (m_exitEventLoop) {
276 return;
277 }
278 } else {
279 LOG_WRN("Received external event in event thread \"%s\" but no external event callback is registered.", m_name);
280 }
281 } else {
282 // It's a function to run in the context of the event thread, run it
283 std::get<std::function<void()>>(msgQueueItem)();
284 }
285 continue;
286 } else if (queueRc == -EAGAIN) {
287 // Queue timed out, which means we need to handle the timer expiry,
288 // jump back to start of while loop to handle the timer expiry
289 LOG_DBG("Queue timed out, which means we need to handle timer expiry.");
290 continue;
291 } else if (queueRc == -ENOMSG) {
292 // This means the queue must have been purged
293 __ASSERT(false, "Got -ENOMSG from queue, was not expecting this.");
294 } else {
295 __ASSERT(false, "Got unexpected return code from queue: %d.", queueRc);
296 }
297 } // End of while (true) loop
298 }
