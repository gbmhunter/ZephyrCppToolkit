134 void runEventLoop() {
135 LOG_MODULE_DECLARE(EventThread, ZCT_EVENT_THREAD_LOG_LEVEL);
136 LOG_DBG("%s() called.", __FUNCTION__);
137
138 while (!m_exitEventLoop) {
139 // Get the next timer to expire
140 auto nextTimerInfo = m_timerManager.getNextExpiringTimer();
141
142 // Check for expired timers and call their callbacks
143 while (nextTimerInfo.m_timer != nullptr && nextTimerInfo.m_durationToWaitUs == 0) {
144 LOG_DBG("Timer expired. Timer: %p.", nextTimerInfo.m_timer);
145 // Update the timer after expiry
146 nextTimerInfo.m_timer->updateAfterExpiry();
147
148 // Call the callback
149 const auto& callback = nextTimerInfo.m_timer->getExpiryCallback();
150 if (callback) {
151 callback();
152 // If the callback calls exitEventLoop(), we will exit the event loop
153 // and return from runEventLoop().
154 if (m_exitEventLoop) {
155 return;
156 }
157 }
158
159 // Check for the next expired timer
160 nextTimerInfo = m_timerManager.getNextExpiringTimer();
161 }
162
163 // If we get here, we have handled all expired timers.
164 k_timeout_t timeout;
165 if (nextTimerInfo.m_timer != nullptr) {
166 timeout = Z_TIMEOUT_US(nextTimerInfo.m_durationToWaitUs);
167 } else {
168 timeout = K_FOREVER;
169 }
170
171 // Block on message queue until next timer expiry
172 // Create storage in the case we receive an external event
173 EventType rxEvent;
174 int queueRc = k_msgq_get(&m_threadMsgQueue, &rxEvent, timeout);
175 if (queueRc == 0) {
176 // We got a message from the queue, call the external event callback
177 if (m_externalEventCallback) {
178 m_externalEventCallback(rxEvent);
179 // If the callback calls exitEventLoop(), we will exit the event loop
180 // and return from runEventLoop().
181 if (m_exitEventLoop) {
182 return;
183 }
184 } else {
185 LOG_WRN("Received external event in event thread \"%s\" but no external event callback is registered.", m_name);
186 }
187 continue;
188 } else if (queueRc == -EAGAIN) {
189 // Queue timed out, which means we need to handle the timer expiry,
190 // jump back to start of while loop to handle the timer expiry
191 LOG_DBG("Queue timed out, which means we need to handle timer expiry.");
192 continue;
193 } else if (queueRc == -ENOMSG) {
194 // This means the queue must have been purged
195 __ASSERT(false, "Got -ENOMSG from queue, was not expecting this.");
196 } else {
197 __ASSERT(false, "Got unexpected return code from queue: %d.", queueRc);
198 }
199 } // End of while (true) loop
200 }
