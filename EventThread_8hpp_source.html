198 void runEventLoop() {
199 LOG_MODULE_DECLARE(EventThread, ZCT_EVENT_THREAD_LOG_LEVEL);
200 LOG_DBG("%s() called.", __FUNCTION__);
201
202 while (!m_exitEventLoop) {
203 // Get the next timer to expire
204 auto nextTimerInfo = m_timerManager.getNextExpiringTimer();
205
206 // Check for expired timers and call their callbacks
207 while (nextTimerInfo.m_timer != nullptr && nextTimerInfo.m_durationToWaitUs == 0) {
208 LOG_DBG("Timer expired. Timer: %p.", nextTimerInfo.m_timer);
209 // Update the timer after expiry
210 nextTimerInfo.m_timer->updateAfterExpiry();
211
212 // Call the callback
213 const auto& callback = nextTimerInfo.m_timer->getExpiryCallback();
214 if (callback) {
215 callback();
216 // If the callback calls exitEventLoop(), we will exit the event loop
217 // and return from runEventLoop().
218 if (m_exitEventLoop) {
219 return;
220 }
221 }
222
223 // Check for the next expired timer
224 nextTimerInfo = m_timerManager.getNextExpiringTimer();
225 }
226
227 // If we get here, we have handled all expired timers.
228 k_timeout_t timeout;
229 if (nextTimerInfo.m_timer != nullptr) {
230 timeout = Z_TIMEOUT_US(nextTimerInfo.m_durationToWaitUs);
231 } else {
232 timeout = K_FOREVER;
233 }
234
235 // Block on message queue until next timer expiry
236 // Create storage in the case we receive an external event
237 EventType rxEvent;
238 int queueRc = k_msgq_get(&m_threadMsgQueue, &rxEvent, timeout);
239 if (queueRc == 0) {
240 // We got a message from the queue, call the external event callback
241 if (m_externalEventCallback) {
242 m_externalEventCallback(rxEvent);
243 // If the callback calls exitEventLoop(), we will exit the event loop
244 // and return from runEventLoop().
245 if (m_exitEventLoop) {
246 return;
247 }
248 } else {
249 LOG_WRN("Received external event in event thread \"%s\" but no external event callback is registered.", m_name);
250 }
251 continue;
252 } else if (queueRc == -EAGAIN) {
253 // Queue timed out, which means we need to handle the timer expiry,
254 // jump back to start of while loop to handle the timer expiry
255 LOG_DBG("Queue timed out, which means we need to handle timer expiry.");
256 continue;
257 } else if (queueRc == -ENOMSG) {
258 // This means the queue must have been purged
259 __ASSERT(false, "Got -ENOMSG from queue, was not expecting this.");
260 } else {
261 __ASSERT(false, "Got unexpected return code from queue: %d.", queueRc);
262 }
263 } // End of while (true) loop
264 }
