113 EventType waitForEvent() {
114 LOG_MODULE_DECLARE(EventThread, ZCT_EVENT_THREAD_LOG_LEVEL);
115 LOG_DBG("%s() called.", __FUNCTION__);
116
117 // Get the next timer to expire
118 auto nextTimerInfo = m_timerManager.getNextExpiringTimer();
119
120 // See if there is an already expired timer, is so, there is no need to block
121 // on the event queue.
122 if (nextTimerInfo.m_timer != nullptr && nextTimerInfo.m_durationToWaitUs == 0) {
123 nextTimerInfo = m_timerManager.getNextExpiringTimer();
124 LOG_DBG("Timer expired. Timer: %p.", nextTimerInfo.m_timer);
125 // Get the timer event
126 nextTimerInfo.m_timer->updateAfterExpiry();
127 return nextTimerInfo.m_timer->getEvent();
128 }
129
130 // If we get here, we have handled all expired timers.
131 k_timeout_t timeout;
132 if (nextTimerInfo.m_timer != nullptr) {
133 timeout = Z_TIMEOUT_US(nextTimerInfo.m_durationToWaitUs);
134 } else {
135 timeout = K_FOREVER;
136 }
137
138 // Block on message queue until next timer expiry
139 // Create storage in the case we receive an external event
140 EventType rxEvent;
141 int queueRc = k_msgq_get(&m_threadMsgQueue, &rxEvent, timeout);
142 if (queueRc == 0) {
143 // We got a message from the queue, so we can handle it
144 return rxEvent;
145 } else if (queueRc == -EAGAIN) {
146 // Queue timed out, which means we need to handle the timer expiry
147 LOG_DBG("Queue timed out, which means we need to handle the timer expiry.");
148 nextTimerInfo.m_timer->updateAfterExpiry();
149 return nextTimerInfo.m_timer->getEvent();
150 } else if (queueRc == -ENOMSG) {
151 // This means the queue must have been purged
152 __ASSERT(false, "Got -ENOMSG from queue, was not expecting this.");
153 } else {
154 __ASSERT(false, "Got unexpected return code from queue: %d.", queueRc);
155 }
156 __ASSERT(false, "Should not get here.");
157 return EventType();
158 }
