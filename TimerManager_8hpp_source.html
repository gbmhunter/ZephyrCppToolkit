94 // Set output to null in case no timer expired

96 uint64_t durationToWaitUs = 0;
97 // Iterate through all registered timers, and find the one that is expiring next (if any)
98 for(uint32_t i = 0; i < this->m_numTimers; i++) {
99 Timer<EventType> * timer = this->m_timers[i];
100 if (timer->isRunning()) {
101 if (expiredTimer == nullptr || timer->getNextExpiryTimeTicks() < expiredTimer->getNextExpiryTimeTicks()) {
102 // LOG_DBG("Setting expired timer to %p.", timer);
103 expiredTimer = timer;
104 }
105 }
106 }
107 LOG_DBG("Expired timer: %p.\n", expiredTimer);
108
109 // Convert the expiry time to a duration from now
110 // Calculate time to wait for next timeout event
111 durationToWaitUs = 0;
112
113 // Ticks is the fundemental resolution that the kernel does operations at
114 int64_t uptime_ticks = k_uptime_ticks();
115 if (expiredTimer != nullptr) {
116 if (expiredTimer->getNextExpiryTimeTicks() <= uptime_ticks) {
117 durationToWaitUs = 0;
118 LOG_DBG("Timer expired.");
119 // Need to update the timer now that we have detected it has expired.
120 // This will either stop the timer if it is a one-shot, or update the next expiry time
121 expiredTimer->updateAfterExpiry();
122 } else {
123 durationToWaitUs = k_ticks_to_us_ceil64(expiredTimer->getNextExpiryTimeTicks() - uptime_ticks);
124 LOG_DBG("Time to wait in us: %llu.", durationToWaitUs);
125 }
126 }

129 LOG_DBG("No timers running.");
130 }
